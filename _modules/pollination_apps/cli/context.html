    def from_file(cls):
        os.makedirs(DEFAULT_CONFIG_DIR, exist_ok=True)
        try:
            return cls.parse_file(DEFAULT_CONFIG_PATH)
        except:
            ctx = cls()
            ctx.save()
            return ctx
